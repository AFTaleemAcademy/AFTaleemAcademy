پہلا کلمہ طیبہ

عربی

لَا إِلٰهَ إِلَّا اللَّهُ مُحَمَّدٌ رَسُولُ اللَّهِ

ترجمہ

اللہ کے سوا کوئی معبود نہیں، محمد ﷺ اللہ کے رسول ہیں۔

دوسرا کلمہ شہادت

عربی

أَشْهَدُ أَنْ لَا إِلٰهَ إِلَّا اللَّهُ وَحْدَهُ لَا شَرِيكَ لَهُ، وَأَشْهَدُ أَنَّ مُحَمَّدًا عَبْدُهُ وَرَسُولُهُ

ترجمہ

میں گواہی دیتا ہوں کہ اللہ کے سوا کوئی معبود نہیں، وہ اکیلا ہے، اس کا کوئی شریک نہیں، اور میں گواہی دیتا ہوں کہ محمد ﷺ اس کے بندے اور رسول ہیں۔

تیسرا کلمہ تمجید

عربی

سُبْحَانَ اللَّهِ وَالْحَمْدُ لِلَّهِ وَلَا إِلٰهَ إِلَّا اللَّهُ وَاللَّهُ أَكْبَرُ وَلَا حَوْلَ وَلَا قُوَّةَ إِلَّا بِاللَّهِ الْعَلِيِّ الْعَظِيمِ

ترجمہ

اللہ پاک ہے، تمام تعریفیں اللہ کے لیے ہیں، اللہ کے سوا کوئی معبود نہیں، اللہ سب سے بڑا ہے، اور گناہ سے بچنے اور نیکی کرنے کی طاقت صرف اللہ ہی سے ہے۔

چوتھا کلمہ توحید

عربی

لَا إِلٰهَ إِلَّا اللَّهُ وَحْدَهُ لَا شَرِيكَ لَهُ، لَهُ الْمُلْكُ وَلَهُ الْحَمْدُ، يُحْيِي وَيُمِيتُ، وَهُوَ حَيٌّ لَا يَمُوتُ أَبَدًا أَبَدًا، ذُو الْجَلَالِ وَالْإِكْرَامِ، بِيَدِهِ الْخَيْرُ، وَهُوَ عَلَى كُلِّ شَيْءٍ قَدِيرٌ

ترجمہ

اللہ کے سوا کوئی معبود نہیں، وہ اکیلا ہے، اس کا کوئی شریک نہیں، اسی کی بادشاہی ہے، اسی کے لیے تمام تعریف ہے، وہی زندہ کرتا ہے اور موت دیتا ہے، وہ ہمیشہ زندہ ہے، اس کے ہاتھ میں ہر بھلائی ہے، اور وہ ہر چیز پر قادر ہے۔

پانچواں کلمہ استغفار

عربی

أَسْتَغْفِرُ اللَّهَ رَبِّي مِنْ كُلِّ ذَنْبٍ أَذْنَبْتُهُ عَمْدًا أَوْ خَطَأً، سِرًّا أَوْ عَلَانِيَةً، وَأَتُوبُ إِلَيْهِ مِنَ الذَّنْبِ الَّذِي أَعْلَمُ وَمِنَ الذَّنْبِ الَّذِي لَا أَعْلَمُ، إِنَّكَ أَنْتَ عَلَّامُ الْغُيُوبِ وَسَتَّارُ الْعُيُوبِ وَغَفَّارُ الذُّنُوبِ، وَلَا حَوْلَ وَلَا قُوَّةَ إِلَّا بِاللَّهِ الْعَلِيِّ الْعَظِيمِ.

ترجمہ

میں اللہ سے، جو میرا رب ہے، اپنے ہر گناہ کی معافی مانگتا ہوں، خواہ وہ جان بوجھ کر کیا ہو یا بھول سے، چھپ کر کیا ہو یا ظاہر میں، اور میں اس کی بارگاہ میں توبہ کرتا ہوں ان گناہوں سے جنہیں میں جانتا ہوں اور ان سے بھی جنہیں میں نہیں جانتا۔ بے شک تو ہی غیبوں کو خوب جاننے والا، عیبوں کو چھپانے والا اور گناہوں کو بہت زیادہ بخشنے والا ہے، اور گناہوں سے بچنے اور نیکی کرنے کی طاقت اللہ بلند و عظمت والے ہی کی طرف سے ہے۔

چھٹا کلمہ ردِّ کفر

عربی

اللَّهُمَّ إِنِّي أَعُوذُ بِكَ مِنْ أَنْ أُشْرِكَ بِكَ شَيْئًا وَأَنَا أَعْلَمُ بِهِ، وَأَسْتَغْفِرُكَ لِمَا لَا أَعْلَمُ بِهِ، تُبْتُ عَنْهُ، وَتَبَرَّأْتُ مِنَ الْكُفْرِ وَالشِّرْكِ وَالْكَذِبِ وَالْغِيبَةِ وَالْبِدْعَةِ وَالنَّمِيمَةِ وَالْفَوَاحِشِ وَالْبُهْتَانِ وَالْمَعَاصِي كُلِّهَا، وَأَسْلَمْتُ وَأَقُولُ لَا إِلٰهَ إِلَّا اللَّهُ مُحَمَّدٌ رَسُولُ اللَّهِ

ترجمہ

اے اللہ! میں تیری پناہ مانگتا ہوں کہ جانتے بوجھتے کسی کو تیرا شریک بناؤں، اور جو لاعلمی میں ہو جائے اس کی معافی مانگتا ہوں۔ میں کفر، شرک، جھوٹ، غیبت، بدعت، چغلی، بے حیائی، بہتان اور تمام گناہوں سے توبہ کرتا ہوں، اور اقرار کرتا ہوں کہ اللہ کے سوا کوئی معبود نہیں اور محمد ﷺ اللہ کے رسول ہیں۔

ایمانِ مجمل

عربی

آمَنْتُ بِاللّٰهِ كَمَا هُوَ بِأَسْمَائِهِ وَصِفَاتِهِ، وَقَبِلْتُ جَمِيعَ أَحْكَامِهِ إِقْرَارًا بِاللِّسَانِ وَتَصْدِيقًا بِالْقَلْبِ.

ترجمہ

میں اللہ تعالیٰ پر ایمان لایا جیسا کہ وہ اپنے ناموں اور صفات کے ساتھ ہے، اور میں نے اس کے تمام احکام کو زبان سے اقرار اور دل سے تصدیق کرتے ہوئے قبول کیا۔

ایمانِ مفصل

عربی

آمَنْتُ بِاللّٰهِ وَمَلَائِكَتِهِ وَكُتُبِهِ وَرُسُلِهِ وَالْيَوْمِ الْآخِرِ وَالْقَدْرِ خَيْرِهِ وَشَرِّهِ مِنَ اللّٰهِ تَعَالَى وَالْبَعْثِ بَعْدَ الْمَوْتِ.

ترجمہ

میں ایمان لایا اللہ پر، اس کے فرشتوں پر، اس کی کتابوں پر، اس کے رسولوں پر، آخرت کے دن پر، اچھی اور بری تقدیر اللہ تعالیٰ کی طرف سے ہونے پر، اور مرنے کے بعد دوبارہ زندہ کیے جانے پر۔

گھر میں داخل ہونے کی دعا

عربی

السَّلاَمُ عَلَيْكُمْ وَرَحْمَةُ اللهِ وَبَرَكَاتُهُ

ترجمہ

"آپ پر سلامتی ہو، اور اللہ کی رحمتیں اور اس کی برکتیں نازل ہوں۔

گھر سے باہر نکلنے کی دعا

عربی

بِسْمِ اللّٰهِ، تَوَكَّلْتُ عَلَى اللّٰهِ، لَا حَوْلَ وَلَا قُوَّةَ إِلَّا بِاللّٰهِ

ترجمہ

اللہ کے نام سے ہم نے اللہ پہ بھروسہ کیا گناہوں سے بچنے کی اور نیکی کرنے کی قوت اللہ ہی کے مدد سے ہے

سونے ک دعا

عربی

اللَّهُمَّ بِاسْمِكَ أَمُوتُ وَأَحْيَا

ترجمہ

اے اللہ ہم تیرے ہی نام سے موت کی نیند سوتا ہوں اور پھر زندہ ہوتا ہوں۔

جاگنے کی دعا

عربی

الْحَمْدُ لِلَّهِ الَّذِي أَحْيَانَا بَعْدَ مَا أَمَاتَنَا وَإِلَيْهِ النُّشُورُ۔

ترجمہ

تمام تعریفیں اس اللہ کے لیے جس نے ہمیں مرنے کے بعد پھر زندہ کیا آور اسی کے طرف پلٹ کر جانا ہے۔

کھانے سے پہلے کی دعا

عربی

بِسْمِ اللَّهِ وَعَلَى بَرَكَةِ اللَّهِ

ترجمہ

میں کھانا شروع کرتا ہوں اللہ کے نام کے ساتھ اور اس کے برکت کے ساتھ۔۔

کھانے کے بعد کی دعا

عربی

الْحَمْدُ لِلَّهِ الَّذِي أَطْعَمَنَا وَسَقَانَا وَجَعَلَنَا مِنَ الْمُسْلِمِينَ

ترجمہ

تمام تعریفیں اس اللہ کے لیے ہیں جس نے ہمیں کھانا کھلایا اور پانی پلایا اور ہمیں مسلمانوں میں سے بنایا۔

بیت الخلاء میں داخل ہونے کی دعا

عربی

اللَّهُمَّ إِنِّي أَعُوذُ بِكَ مِنَ الْخُبُثِ وَالْخَبَائِثِ

ترجمہ

اللہ کے نام کے ساتھ۔ اے اللہ! میں تمام ناپاک چیزوں اور خبیث جنات (نر اور مادہ) سے تیری پناہ مانگتا ہوں

بیت الخلاء سے نکلنے کی دعا

عربی

غُفْرَانَكَ +.الْحَمْدُ لِلَّهِ الَّذِي أَذْهَبَ عَنِّي الْأَذَى وَعَافَانِي

ترجمہ

تمام تعریفیں اللہ کے لیے ہیں جس نے مجھ سے تکلیف (گندگی) کو دور کیا اور مجھے عافیت (صحت) بخشی۔

سفر کی دعا

عربی

سُبْحَانَ الَّذِي سَخَّرَ لَنَا هَٰذَا وَمَا كُنَّا لَهُ مُقْرِنِينَ وَإِنَّا إِلَىٰ رَبِّنَا لَمُنْقَلِبُونَ

ترجمہ

پاک ہے وہ ذات جس نے اس (سواری) کو ہمارے بس میں کر دیا، حالانکہ ہم اسے قابو میں کرنے کی طاقت نہیں رکھتے تھے، اور بلاشبہ ہم اپنے رب کی طرف لوٹ کر جانے والے ہیں۔"

والدین کے لیے دعا

عربی

رَّبِّ ارْحَمْهُمَا كَمَا رَبَّيَانِي صَغِيرًا

ترجمہ

اے میرے رب! ان دونوں (والدین) پر رحم فرما، جیسے انہوں نے بچپن میں میری پرورش کی تھی۔"

وضو سے پہلے کی دعا

عربی

بِسْمِ اللَّهِ

ترجمہ

اللہ کے نام سے شروع کرتا ہوں۔

وضو کے بعد کی دعا

عربی

أَشْهَدُ أَنْ لَا إِلٰهَ إِلَّا اللَّهُ وَحْدَهُ لَا شَرِيكَ لَهُ، وَأَشْهَدُ أَنَّ مُحَمَّدًا عَبْدُهُ وَرَسُولُهُ

ترجمہ

میں گواہی دیتا ہوں کہ اللہ کے سوا کوئی معبود نہیں، وہ اکیلا ہے، اس کا کوئی شریک نہیں، اور محمد ﷺ اس کے بندے اور رسول ہیں۔

مسجد میں داخل ہونے کی دعا

عربی

اللَّهُمَّ افْتَحْ لِي أَبْوَابَ رَحْمَتِكَ

ترجمہ

اے اللہ! میرے لیے اپنی رحمت کے دروازے کھول دے۔

مسجد سے نکلنے کی دعا

عربی

اللَّهُمَّ إِنِّي أَسْأَلُكَ مِنْ فَضْلِكَ

ترجمہ

اے اللہ! میں تجھ سے تیرے فضل کا سوال کرتا ہوں۔

آئینہ دیکھنے کی دعا

عربی

اللَّهُمَّ كَمَا أَحْسَنْتَ خَلْقِي فَأَحْسِنْ خُلُقِي

ترجمہ

اے اللہ! جیسے تو نے میری صورت اچھی بنائی ہے، ویسے ہی میرا اخلاق بھی اچھا بنا۔

نئے کپڑے پہننے کی دعا

عربی

الْحَمْدُ لِلَّهِ الَّذِي كَسَانِي هَذَا

ترجمہ

تمام تعریف اللہ کے لیے ہے جس نے مجھے یہ لباس عطا فرمایا۔

کپڑے اتارنے کی دعا

عربی

بِسْمِ اللَّهِ

ترجمہ

اللہ کے نام سے۔

چھینک آنے پر دعا

عربی

الْحَمْدُ لِلَّهِ

ترجمہ

تمام تعریف اللہ ہی کے لیے ہے۔

چھینک کا جواب

عربی

يَرْحَمُكَ اللَّهُ

ترجمہ

اللہ تم پر رحم فرمائے۔

بیمار کی عیادت کی دعا

عربی

لَا بَأْسَ طَهُورٌ إِنْ شَاءَ اللَّهُ

ترجمہ

ان شاء اللہ کوئی حرج نہیں، یہ بیماری گناہوں کی پاکیزگی کا سبب ہے۔

شفا کی دعا

عربی

اللَّهُمَّ اشْفِهِ

ترجمہ

اے اللہ! اسے شفا عطا فرما۔

غصہ آنے کی دعا

عربی

أَعُوذُ بِاللَّهِ مِنَ الشَّيْطَانِ الرَّجِيمِ

ترجمہ

میں شیطان مردود سے اللہ کی پناہ مانگتا ہوں۔

بازار میں داخل ہونے کی دعا

عربی

لَا إِلٰهَ إِلَّا اللَّهُ وَحْدَهُ لَا شَرِيكَ لَهُ...

ترجمہ

اللہ کے سوا کوئی معبود نہیں، وہ اکیلا ہے، اس کا کوئی شریک نہیں۔

بارش کی دعا

عربی

اللَّهُمَّ صَيِّبًا نَافِعًا

ترجمہ

اے اللہ! اس بارش کو نفع والی بنا۔

اذان سننے کے بعد کی دعا

عربی

اللَّهُمَّ رَبَّ هَذِهِ الدَّعْوَةِ التَّامَّةِ...

ترجمہ

اے اللہ! اس کامل دعوت اور قائم ہونے والی نماز کے رب۔

استخارہ کی دعا

عربی

اللَّهُمَّ إِنِّي أَسْتَخِيرُكَ بِعِلْمِكَ...

ترجمہ

اے اللہ! میں تیرے علم کے ذریعے تجھ سے خیر طلب کرتا ہوں۔

قرض سے بچنے کی دعا

عربی

اللَّهُمَّ إِنِّي أَعُوذُ بِكَ مِنَ الْمَأْثَمِ وَالْمَغْرَمِ

ترجمہ

اے اللہ! میں گناہ اور قرض سے تیری پناہ مانگتا ہوں۔

رزق میں برکت کی دعا

عربی

اللَّهُمَّ بَارِكْ لَنَا فِيمَا رَزَقْتَنَا

ترجمہ

اے اللہ! ہمیں عطا کیے ہوئے رزق میں برکت عطا فرما۔

علم بڑھنے کی دعا

عربی

رَبِّ زِدْنِي عِلْمًا

ترجمہ

اے میرے رب! میرے علم میں اضافہ فرما۔

مشکل آسان ہونے کی دعا

عربی

رَبِّ يَسِّرْ وَلَا تُعَسِّرْ وَتَمِّمْ بِالْخَيْرِ

ترجمہ

اے میرے رب! آسانی فرما، مشکل نہ فرما، اور خیر کے ساتھ مکمل فرما۔

بخار کی دعا

عربی

اللَّهُمَّ اشْفِنِي شِفَاءً لَا يُغَادِرُ سَقَمًا

ترجمہ

اے اللہ! مجھے ایسی شفا عطا فرما کہ کوئی بیماری باقی نہ رہے۔

مریض کی عیادت کی دعا

عربی

لَا بَأْسَ طَهُورٌ إِنْ شَاءَ اللَّهُ

ترجمہ

کوئی حرج نہیں، ان شاء اللہ یہ بیماری پاکیزگی کا سبب ہے۔

چاند دیکھنے کی دعا

عربی

اللَّهُمَّ أَهِلَّهُ عَلَيْنَا بِالْأَمْنِ وَالْإِيمَانِ وَالسَّلَامَةِ وَالْإِسْلَامِ

ترجمہ

اے اللہ! اس چاند کو ہمارے لیے امن، ایمان، سلامتی اور اسلام کے ساتھ طلوع فرما۔

آندھی کی دعا

عربی

اللَّهُمَّ إِنِّي أَسْأَلُكَ خَيْرَهَا وَأَعُوذُ بِكَ مِنْ شَرِّهَا

ترجمہ

اے اللہ! میں تجھ سے اس کی بھلائی مانگتا ہوں اور اس کے شر سے پناہ مانگتا ہوں۔

گرج چمک کی دعا

عربی

سُبْحَانَ الَّذِي يُسَبِّحُ الرَّعْدُ بِحَمْدِهِ

ترجمہ

پاک ہے وہ ذات جس کی حمد کے ساتھ گرج تسبیح کرتی ہے۔

قبرستان میں داخل ہونے کی دعا

عربی

السَّلَامُ عَلَيْكُمْ أَهْلَ الدِّيَارِ مِنَ الْمُؤْمِنِينَ وَالْمُسْلِمِينَ

ترجمہ

اے مومنو اور مسلمانو! تم پر سلام ہو۔

نظر بد سے حفاظت کی دعا

عربی

أَعُوذُ بِكَلِمَاتِ اللَّهِ التَّامَّاتِ مِنْ شَرِّ مَا خَلَقَ

ترجمہ

میں اللہ کے کامل کلمات کے ذریعے ہر مخلوق کے شر سے پناہ مانگتا ہوں۔

غم اور پریشانی کی دعا

عربی

حَسْبِيَ اللَّهُ لَا إِلٰهَ إِلَّا هُوَ عَلَيْهِ تَوَكَّلْتُ

ترجمہ

اللہ مجھے کافی ہے، اس کے سوا کوئی معبود نہیں، اسی پر میں نے بھروسہ کیا۔

قرض ادا ہونے کی دعا

عربی

اللَّهُمَّ اكْفِنِي بِحَلَالِكَ عَنْ حَرَامِكَ وَأَغْنِنِي بِفَضْلِكَ عَمَّنْ سِوَاكَ

ترجمہ

اے اللہ! اپنے حلال کے ذریعے مجھے حرام سے بچا اور اپنے فضل سے مجھے دوسروں سے بے نیاز کر دے۔

ہدایت کی دعا

عربی

رَبَّنَا لَا تُزِغْ قُلُوبَنَا بَعْدَ إِذْ هَدَيْتَنَا

ترجمہ

اے ہمارے رب! ہدایت دینے کے بعد ہمارے دلوں کو ٹیڑھا نہ کر۔
